/**
 * Tabula.js models and runs the Roman board game Tabula.
 *
 * The module exposes the game state, legal move calculation, turn
 * progression, captures, bar re-entry, bearing off, and win detection for
 * the standard 24-point Tabula board.
 *
 * @namespace Tabula
 * @author Emily Kanagasabay
 * @version 2026
 */
const Tabula = Object.create(null);

/**
 * The full immutable-style game state for Tabula.
 *

 * @property {Tabula.Player} currentPlayer Player whose turn is active.

/**
 * Returns the opponent of the given player.
 *
 * @function
 * @ignore
 * @param {Tabula.Player} player Player attempting the move or selection.
 * @returns {Tabula.Player}
 */
const getOpponent = function (player) {
    if (player === Player.WHITE) {
        return Player.BLACK;
    }
    return Player.WHITE;
};

/**
 * Returns movement direction for a player.
 * WHITE moves forward (+1), BLACK moves backward (-1).
 *
 * @function
 * @ignore
 * @param {Tabula.Player} player Player attempting the move or selection.
 * @returns {number}
 */
const getDirection = function (player) {
    if (player === Player.WHITE) {
        return 1;
    }
    return -1;
};

/**
 * Checks whether a board index is valid.
 *
 * @function
 * @ignore
 * @param {number} index Board index to validate or access.
 * @returns {boolean}
 */
const isWithinBoard = (index) => index >= 0 && index < 24;

/**
 * Safely retrieves a board point.
 *
 * @function
 * @ignore
 * @param {Tabula.Point[]} board Board array representing points and piece
 *     counts.
 * @param {number} index Board index to validate or access.
 * @returns {Tabula.Point|null}
 */
const getBoardPoint = function (board, index) {
    if (isWithinBoard(index)) {
        return board[index];
    }
    return null;
};

/**
 * Creates a new game state.
 * WHITE always starts, no dice are active yet, and both bars/off-board
 * counts start at zero.
 *
 * @example
 * let game = Tabula.createGame();
 * game = Tabula.rollDice(game);
 *
 * const from = 0;
 * const to = Tabula.getLegalMoves(game, from)[0];
 *
 * if (to !== undefined && Tabula.isLegalMove(game, { from, to })) {
 *     game = Tabula.movePiece(game, { from, to });
 * }
 *
 * if (Tabula.isTurnComplete(game)) {
 *     game = Tabula.switchPlayer(game);
 * }
 *
 * @function
 * @memberof Tabula
 * @returns {Tabula.GameState} Initial game state
 */
Tabula.createGame = () => ({
    board: createInitialBoard(),
    bar: {
        WHITE: 0,
        BLACK: 0
    },
    borneOff: {
        WHITE: 0,
        BLACK: 0
    },
    currentPlayer: Player.WHITE,
    dice: [],
    selectedPoint: null,
    turn: 1,
    winner: null
});

/**
 * Rolls a single die (1–6).
 *
 * @function
 * @ignore
 * @returns {number}
 */
const rollDie = () => Math.floor(Math.random() * 6) + 1;

/**
 * Roll three dice for the current turn.
 *
 * Has no effect if the game is already won or dice are already available.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @returns {Tabula.GameState}
 */
Tabula.rollDice = function (state) {
    if (state.winner || state.dice.length > 0) {
        return state;
    }

    return Object.assign({}, state, {
        dice: [rollDie(), rollDie(), rollDie()]
    });
};

/**
 * Returns entry index for bar entry.
 *
 * @function
 * @ignore
 * @param {Tabula.Player} player Player attempting the move or selection.
 * @returns {number}
 */
const getBarEntryIndex = function (player) {
    if (player === Player.WHITE) {
        return 0;
    }
    return 23;
};

/**
 * Determines whether a destination is open.
 *
 * @function
 * @ignore
 * @param {Tabula.Point} point Board point to inspect.
 * @param {Tabula.Player} player Player attempting the move or selection.
 * @returns {boolean}
 */
const isDestinationOpen = function (point, player) {
    return (
        point.player === player ||
        point.player === null ||
        point.count === 1
    );
};

/**
 * Determines if a piece can be borne off.
 *
 * @function
 * @ignore
 * @param {Tabula.GameState} state Current game state.
 * @param {number} pointIndex Point index to evaluate for
 * movement or bearing off.
 * @param {number} dieValue Die value to match exactly against
 * the distance from the point to the board edge.
 * @returns {boolean}
 */
const canBearOffFrom = function (state, pointIndex, dieValue) {
    const board = state.board;
    const currentPlayer = state.currentPlayer;

    let homeStart;
    let homeEnd;

    if (currentPlayer === Player.WHITE) {
        homeStart = 18;
        homeEnd = 23;
    } else {
        homeStart = 0;
        homeEnd = 5;
    }

    const inHomeBoard = board.reduce(function (sum, point, index) {
        if (point.player !== currentPlayer) {
            return sum;
        }

        if (currentPlayer === Player.WHITE) {
            if (index >= homeStart) {
                return sum + point.count;
            }
            return sum;
        }

        if (index <= homeEnd) {
            return sum + point.count;
        }
        return sum;
    }, 0);

    const totalOnBoard = board.reduce(
        function (sum, p) {
            if (p.player === currentPlayer) {
                return sum + p.count;
            }
            return sum;
        },
        0
    );

    if (totalOnBoard === 0 || inHomeBoard !== totalOnBoard) {
        return false;
    }

    let distance;

    if (currentPlayer === Player.WHITE) {
        distance = 24 - pointIndex;
    } else {
        distance = pointIndex + 1;
    }

    return dieValue === distance;
};

/**
 * Generates all permutations of an array.
 *
 * @ignore
 * @param {number[]} values Dice values to generate sequences from.
 * @returns {number[][]}
 */
const getPermutations = function (values) {
    if (values.length <= 1) {
        return [values];
    }

    return values.flatMap(function (value, index) {
        const remaining = [
            ...values.slice(0, index),
            ...values.slice(index + 1)
        ];

        return getPermutations(remaining).map(
            function (perm) {
                return [value, ...perm];
            }
        );
    });
};

/**
 * Generates all ordered non-empty dice sequences.
 *
 * @ignore
 * @param {number[]} values Dice values to generate sequences from.
 * @returns {number[][]}
 */
const getDiceSequences = function (values) {
    const results = [];

    const build = function (
        remaining,
        current
    ) {
        if (current.length > 0) {
            results.push(
                ...getPermutations(current)
            );
        }

        remaining.forEach(function (value, index) {
            build(
                [
                    ...remaining.slice(0, index),
                    ...remaining.slice(index + 1)
                ],
                [...current, value]
            );
        });
    };

    build(values, []);

    return results;
};

/**
 * Checks whether every step in a movement sequence is legal.
 *
 * @ignore
 * @param {Tabula.Point[]} board Board points.
 * @param {Tabula.Player} currentPlayer Player attempting the sequence.
 * @param {number} start Starting board index for the sequence.
 * @param {number[]} sequence Ordered dice sequence to follow.
 * @returns {number|null}
 */
const followSequence = function (
    board,
    currentPlayer,
    start,
    sequence
) {
    const direction = getDirection(currentPlayer);

    let current = start;

    const isValid = sequence.every(function (die) {
        const destination = current + direction * die;

        if (!isWithinBoard(destination)) {
            return false;
        }

        const point = getBoardPoint(board, destination);

        if (
            !isDestinationOpen(
                point,
                currentPlayer
            )
        ) {
            return false;
        }

        current = destination;
        return true;
    });

    if (!isValid) {
        return null;
    }

    return current;
};

/**
 * Returns every legal destination plus the dice sequence used.
 *
 * @ignore
 * @param {Tabula.GameState} state Current game state.
 * @param {number|Tabula.Location} pointIndex Source point index or BAR.
 * @returns {Tabula.LegalMoveOption[]}
 */
const getLegalMoveOptions = function (state, pointIndex) {
    const board = state.board;
    const currentPlayer = state.currentPlayer;
    const dice = state.dice;
    const bar = state.bar;

    if (dice.length === 0) {
        return [];
    }

    if (pointIndex === Location.BAR) {
        if (bar[currentPlayer] === 0) {
            return [];
        }
    } else {
        const sourcePoint = getBoardPoint(board, pointIndex);

        if (
            !sourcePoint ||
            sourcePoint.player !== currentPlayer ||
            sourcePoint.count === 0
        ) {
            return [];
        }

        if (bar[currentPlayer] > 0) {
            return [];
        }
    }

    let startIndex;

    if (pointIndex === Location.BAR) {
        startIndex = getBarEntryIndex(currentPlayer);
    } else {
        startIndex = pointIndex;
    }

    const sequences = getDiceSequences(dice);

    const legalOptions = [];

    sequences.forEach(function (sequence) {
        const destination = followSequence(
            board,
            currentPlayer,
            startIndex,
            sequence
        );

        if (destination !== null) {
            legalOptions.push({
                to: destination,
                sequence
            });
        }

        const totalDistance = sequence.reduce(
            function (sum, value) {
                return sum + value;
            },
            0
        );

        if (
            pointIndex !== Location.BAR &&
            canBearOffFrom(
                state,
                pointIndex,
                totalDistance
            )
        ) {
            legalOptions.push({
                to: Location.OFF,
                sequence
            });
        }
    });

    return legalOptions;
};

/**
 * Removes an ordered sequence of dice.
 *
 * @ignore
 * @param {number[]} availableDice Dice values still available to use.
 * @param {number[]} usedDice Dice values already consumed for a move.
 * @returns {number[]}
 */
const removeDiceSequence = function (
    availableDice,
    usedDice
) {
    const remaining = availableDice.slice();

    usedDice.forEach(function (die) {
        const index = remaining.indexOf(die);

        if (index !== -1) {
            remaining.splice(index, 1);
        }
    });

    return remaining;
};

/**
 * Selects a deterministic dice sequence for a move.
 *
 * @ignore
 * @param {Tabula.LegalMoveOption[]} options
 * Candidate legal options for a single destination.
 * @param {number[]|undefined} explicitPath Optional explicit sequence.
 * @returns {number[]|null}
 */
const selectMoveSequence = function (options, explicitPath) {
    if (options.length === 0) {
        return null;
    }

    if (explicitPath !== undefined) {
        if (!Array.isArray(explicitPath) || explicitPath.length === 0) {
            return null;
        }

        const exactMatch = options.find(function (option) {
            return R.equals(option.sequence, explicitPath);
        });

        if (!exactMatch) {
            return null;
        }
        return exactMatch.sequence;
    }

    const compareSequence = function (left, right) {
        if (left.length !== right.length) {
            return left.length - right.length;
        }

        const count = Math.min(left.length, right.length);
        let i = 0;

        while (i < count) {
            if (left[i] !== right[i]) {
                return left[i] - right[i];
            }
            i += 1;
        }

        return 0;
    };

    const sorted = options.slice().sort(function (left, right) {
        return compareSequence(
            left.sequence,
            right.sequence
        );
    });

    return sorted[0].sequence;
};

/**
 * Return every legal destination reachable from a source point using the
 * remaining dice in any valid order.
 * Intermediate steps must also be legal, and bar re-entry takes priority
 * over normal board movement.
 *
 * Domain rules applied:
 * - If the current player has pieces on the bar, only BAR can be used as
 *   the source.
 * - Destinations occupied by two or more opponent pieces are blocked.
 * - A destination can be reached via a multi-die path only when each
 *   intermediate step is legal.
 * - OFF is included only when bearing-off rules are satisfied.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @param {number|Tabula.Location} pointIndex Source point index or
 * special location for the move.
 * @returns {Array.<number|Tabula.Location>}
 */
Tabula.getLegalMoves = function (state, pointIndex) {
    const legalOptions = getLegalMoveOptions(state, pointIndex);

    return R.uniq(legalOptions.map(function (option) {
        return option.to;
    }));
};

/**
 * Check whether a proposed move is legal in the current position.
 * Returns false for malformed moves, blocked destinations,
 * bar-priority violations, or moves that cannot be made with the
 * available dice.
 *
 * If move.dicePath is omitted, any legal dice sequence for that
 * source/destination pair is accepted.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @param {Tabula.Move} move Proposed move with source and destination.
 * @returns {boolean}
 */
Tabula.isLegalMove = function (state, move) {
    if (!move || typeof move !== "object") {
        return false;
    }

    const options = getLegalMoveOptions(
        state,
        move.from
    ).filter(function (option) {
        return option.to === move.to;
    });

    if (options.length === 0) {
        return false;
    }

    if (move.dicePath === undefined) {
        return true;
    }

    if (!Array.isArray(move.dicePath) || move.dicePath.length === 0) {
        return false;
    }

    return options.some(function (option) {
        return R.equals(option.sequence, move.dicePath);
    });
};

/**
 * Execute one legal Tabula move and return the next game state.
 *
 * Domain effects:
 * - Captures a single opposing piece by sending it to the bar.
 * - Removes only the dice values consumed by this move.
 * - Clears selectedPoint after the move resolves.
 * - Sets winner when the active player has borne off all 15 pieces.
 *
 * If the move is not legal, the original state is returned unchanged.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @param {Tabula.Move} move Proposed move with source and destination.
 * @returns {Tabula.GameState}
 */
Tabula.movePiece = function (state, move) {
    if (state.winner || !move || typeof move !== "object") {
        return state;
    }

    const legalOptions = getLegalMoveOptions(
        state,
        move.from
    ).filter(function (option) {
        return option.to === move.to;
    });

    const usedDice = selectMoveSequence(
        legalOptions,
        move.dicePath
    );

    if (!usedDice) {
        return state;
    }

    const board = state.board;
    const currentPlayer = state.currentPlayer;
    const bar = state.bar;
    const borneOff = state.borneOff;
    const opponent = getOpponent(currentPlayer);

    const nextBoard = board.map(function (p) {
        return {
            player: p.player,
            count: p.count
        };
    });
    const nextBar = Object.assign({}, bar);
    const nextBorneOff = Object.assign({}, borneOff);

    const sourceIndex = move.from;
    const destination = move.to;

    if (sourceIndex === Location.BAR) {
        nextBar[currentPlayer] -= 1;
    } else {
        const source = nextBoard[sourceIndex];
        const count = source.count - 1;
        let nextPlayer = null;

        if (count > 0) {
            nextPlayer = currentPlayer;
        }

        nextBoard[sourceIndex] = {
            player: nextPlayer,
            count
        };
    }

    if (destination === Location.OFF) {
        nextBorneOff[currentPlayer] += 1;
    } else {
        const target = nextBoard[destination];

        if (target.player === opponent && target.count === 1) {
            nextBar[opponent] += 1;
            nextBoard[destination] = {
                player: currentPlayer,
                count: 1
            };
        } else {
            nextBoard[destination] = {
                player: currentPlayer,
                count: target.count + 1
            };
        }
    }

    const nextDice = removeDiceSequence(state.dice, usedDice);


    let nextWinner = null;

    if (nextBorneOff[currentPlayer] === 15) {
        nextWinner = currentPlayer;
    }

    return Object.assign({}, state, {
        board: nextBoard,
        bar: nextBar,
        borneOff: nextBorneOff,
        dice: nextDice,
        selectedPoint: null,
        winner: nextWinner
    });
};

/**
 * Check whether the active turn is finished.
 *
 * A turn is complete when no dice remain, or when remaining dice cannot
 * produce any legal move from required sources.
 *
 * Bar priority is enforced: if the active player has pieces on the bar,
 * only bar re-entry is considered when determining completion.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @returns {boolean}
 */
Tabula.isTurnComplete = function (state) {
    if (state.dice.length === 0) {
        return true;
    }

    let sources;

    if (state.bar[state.currentPlayer] > 0) {
        sources = [Location.BAR];
    } else {
        sources = state.board.reduce(function (acc, p, i) {
            if (
                p.player === state.currentPlayer &&
                p.count > 0
            ) {
                return acc.concat([i]);
            }
            return acc;
        }, []);
    }

    return !R.any(
        function (s) {
            return Tabula.getLegalMoves(state, s).length > 0;
        },
        sources
    );
};

/**
 * Switch the turn to the opponent and reset turn-local state.
 *
 * Clears remaining dice and current selection, then increments the turn
 * counter.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @returns {Tabula.GameState}
 */
Tabula.switchPlayer = function (state) {
    if (state.winner) {
        return state;
    }

    return Object.assign({}, state, {
        currentPlayer: getOpponent(state.currentPlayer),
        dice: [],
        selectedPoint: null,
        turn: state.turn + 1
    });
};

/**
 * Check whether the game has been won.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @returns {boolean}
 */
Tabula.isGameOver = function (state) {
    return state.winner !== null;
};

/**
 * Read the winning player, if the game is over.
 *
 * @function
 * @memberof Tabula
 * @param {Tabula.GameState} state Current game state.
 * @returns {Tabula.Player|null}
 */
Tabula.getWinner = function (state) {
    return state.winner;
};
